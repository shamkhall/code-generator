according to this table, generate postgres find all functions with limit offset
parameter names has to be like: _param_name and use alias with columns

CREATE TABLE IF NOT EXISTS public.favorite_articles (
  favorite_article_id SERIAL PRIMARY KEY,
  article_id INTEGER NOT NULL references public.articles (article_id),
  user_id INTEGER NOT NULL references public_auth.users (user_id),
  pet_profile_id INTEGER NOT NULL references public.pet_profiles (pet_profile_id)
);

like this:
CREATE OR REPLACE FUNCTION function-name()
RETURNS TABLE (
    ...
)
AS $$
BEGIN
    RETURN QUERY SELECT use alias with columns
                 FROM table-name alias;
END;
$$
LANGUAGE plpgsql;
